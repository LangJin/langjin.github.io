前言 #

openpyxl是python的一个用于操作excle文件的第三方包。

openpyxl相对于已经经常用的xlrd、xlwt等操作的excle的第三方包来说，功能更加的丰富，不仅可以对excle进行读写，还能在excle中创建图表，设置内容的排版和样式。

不过注意的是，openpyxl不支持老版本的excle文件，仅支持最新的xlxs格式。

安装 #

openpyxl的安装很简单，直接利用pip工具安装即可。

shell
pip install openpyxl
# 或者使用豆瓣源进行安装
pip install openpyxl -i https://pypi.douban.com/simple

基本操作 #

名词解释：

  • workbook：excle文件
  • worksheet：sheet表单
  • rows：行
  • column：列
  • cell：单元格

快速上手 #

python
from openpyxl import Workbook,load_workbook

excle = Workbook()	# 初始化一个excle文件对象
sheet = excle.active	# 获取当前文件的sheet对象，一般默认为sheet1
sheet['A1'] = '使用openpyxl操作excle'	# 在sheet1的A1单元格写入内容
excle.save('test.xlsx')	# 保存文件到本地

excle = load_workbook('test.xlsx')	# 加载本地excle文件
sheet = excle.active	# 获取当前文件的sheet对象，一般默认为sheet1
value = sheet['A1'].value	# 根据单元格坐标获取单元格的值
print(value)

操作Sheet #

python
from openpyxl import Workbook

excle = Workbook()
"""
获取默认sheet对象
"""
sheet = excle.active    # 获取默认的sheet，为excle文件中的第一个sheet
sheet.title = '默认表格'    # 修改当前sheet的名称
title = sheet.title     # 获取当前sheet的名字
print(title)
"""
新建sheet对象
create_sheet方法的参数不是必填，
当都不填的时候新建的sheet默认追加在最后，且名字为sheet1,sheet2这种模式。
index是新建的sheet的位置
title是新建的sheet的title
create_sheet完成新建后会返回新建的sheet的对象。
"""
sheet1 = excle.create_sheet(index=1,title='表1') # 创建一个新的sheet，并返回sheet对象
title = sheet1.title
print(title)
"""
获取sheet对象
"""
for sheet in excle: # 遍历sheet对象
    print(sheet.title)
sheetNameList = excle.sheetnames    # 获取所有的sheet对象的title
print(sheetNameList)
sheet1 = excle.get_sheet_by_name("表1") # 根据sheet的title获取sheet对象
sheet1 = excle['表1']   # 根据sheet的title获取sheet对象

"""
复制sheet对象
"""
excle.copy_worksheet(excle['表1'])  # 表1为sheet的title
"""
删除sheet对象
"""
excle.remove(excle['表1'])  # 表1为sheet的title
"""
保存文件
"""
excle.save('测试.xlsx')

操作数据 #

注意：在excle表中写入数据的时候，如果表中没有数据，会按顺序一行一行的自动往后面逐个写入。如果原来存在数据，对应的单元格的数据会被覆盖。

python
from openpyxl import Workbook
from openpyxl.drawing.image import Image
excle = Workbook()
sheet = excle.active
"""
在单元格中写入数据
"""
sheet['A1'] = '通过单元格名字写入数据'
sheet.cell(1,2).value = '通过单元格坐标写入数据'
sheet.cell(row=1, column=2, value="通过cell方法写入数据")
sheet.append(['张三','23岁','男'])	# 按行写入数据
"""
在单元格中插入图片
"""
img = Image('logo.png')  # 导入需要插入的图片
ws.add_image(img, 'A5') # 插入图片到A5
"""
在单元格中插入公式
"""
sheet['A2'] = "=SUM(A1:A3)" # A1和A3的值的和
sheet['A2'] = "=SUM(12,11)" # 12和11的和
"""
获取单元格属性
"""
rowNum = sheet.max_row # 总行数
colNum = sheet.max_column # 总列数
data = sheet['A1']  # 使用单元格名称获取单元格对象
data = sheet.cell(1,2)  # 使用单元格坐标获取单元格对象
row = data.row  # 获取单元格的行
col = data.column   #获取单元格的列
value = data.value  # 获取单元格的值


"""
行的操作
"""
rowList = sheet[1] # 根据行号获取整行对象
rowList = sheet[1:5] # 根据行号获取1-5行对象
rowList = sheet.iter_rows() # 获取每一行的对象（迭代器）
rowList = sheet.iter_rows(min_row=2, max_row=3, min_col=2, max_col=2) # 获取指定范围的行对象（迭代器）
for row in rowList:
    for data in row:
        print(data.value)

sheet.insert_rows(2)    # 在第二行插入一个空行
sheet.delete_rows(2)    # 删除第2行

"""
列的操作
""" 
colList = sheet['A']   # 根据列名获取整列对象
colList = sheet['A':'D']   # 根据列名获取A-D列对象
colList = sheet.iter_cols() # 获取每一列的对象（迭代器）
colList = sheet.iter_cols(min_row=2, max_row=3, min_col=2, max_col=2) # 获取指定范围的列对象（迭代器）
for col in colList:
    for data in col:
        print(data.value)

sheet.insert_cols(2)    # 在第二列插入一列
sheet.delete_cols(2)    # 删除第二列


"""
保存文件
"""
excle.save('测试.xlsx')

操作样式 #

python
from openpyxl import Workbook
from openpyxl.styles import colors, Font,Alignment,Border, Side

excle = openpyxl.Workbook()
sheet = excle.active
# 修改单元格字体样式
c3_font = Font(
    name="宋体",  # 字体
    size=15,         # 字体大小
    color="0000FF",  # 字体颜色，用16进制rgb表示
    bold=True,       # 是否加粗，True/False
    italic=True,     # 是否斜体，True/False
    strike=None,     # 是否使用删除线，True/False
    underline=None,  # 下划线, 可选'singleAccounting', 'double', 'single', 'doubleAccounting'
)
sheet['C3'].font = c3_font
sheet['C3'] = '宋体 24 italic'
a5_font = Font('微软雅黑', size=18, bold=True, color=colors.BLUE)
sheet['A5'].font = a5_font
sheet['A5'] = '微软雅黑 18 bold blue'


# 设置行高和列宽
sheet['A1'] = 'Tall row' 
sheet.row_dimensions[1].height = 70 # 行高
sheet['B2'] = 'Wide column' 
sheet.column_dimensions['B'].width = 20 # 列宽

# 设置对齐方式
sheet['B2'].alignment = Alignment(
    horizontal='left',     # 水平对齐，可选general、left、center、right、fill、justify、centerContinuous、distributed
    vertical='top',        # 垂直对齐， 可选top、center、bottom、justify、distributed
    text_rotation=0,       # 字体旋转，0~180整数
    wrap_text=False,       # 是否自动换行
    shrink_to_fit=False,   # 是否缩小字体填充
    indent=0,              # 缩进值
)

# 设置单元格边框
side = Side(
    style="medium",  # 边框样式，可选dashDot、dashDotDot、dashed、dotted、double、hair、medium、mediumDashDot、mediumDashDotDot、mediumDashed、slantDashDot、thick、thin
    color="ff66dd",  # 边框颜色，16进制rgb表示
)
 
sheet["B2"].border = Border(
    top=side,     # 上
    bottom=side,  # 下
    left=side,    # 左
    right=side,   # 右
    diagonal=side # 对角线
)


# 设置单元格的填充和渐变
fill = PatternFill(
    patternType="solid",  # 填充类型，可选none、solid、darkGray、mediumGray、lightGray、lightDown、lightGray、lightGrid
    fgColor="F562a4",     # 前景色，16进制rgb
    bgColor="0000ff",     # 背景色，16进制rgb
    # fill_type=None,     # 填充类型
    # start_color=None,   # 前景色，16进制rgb
    # end_color=None      # 背景色，16进制rgb
)
sheet["B2"].fill = fill
sheet["B3"].fill = GradientFill(
    degree=60,  # 角度
    stop=("000000", "FFFFFF")  # 渐变颜色，16进制rgb
)

# 合并单元格 
sheet.merge_cells('A1:D3') 
sheet['A1'] = 'Twelve cells merged together' 
sheet.merge_cells('C5:D5') 
sheet['C5'] ='Two merged cells

# 拆分单元格
sheet.unmerge_cells('A1:D3') 
sheet.unmerge_cells('C5:D5') 
  
excle.save('测试.xlsx')

创建表格 #

表格是对单元格组的引用。这使得某些操作（如设置表中单元格的样式）变得更容易。

python
from openpyxl import Workbook
from openpyxl.worksheet.table import Table, TableStyleInfo

wb = Workbook()
ws = wb.active
data = [
    ['Apples', 10000, 5000, 8000, 6000],
    ['Pears',   2000, 3000, 4000, 5000],
    ['Bananas', 6000, 6000, 6500, 6000],
    ['Oranges',  500,  300,  200,  700],
]
ws.append(["Fruit", "2011", "2012", "2013", "2014"])
for row in data:
    ws.append(row)
tab = Table(displayName="Table1", ref="A1:E5")
style = TableStyleInfo(name="TableStyleMedium9", showFirstColumn=False,showLastColumn=False, showRowStripes=True, showColumnStripes=True)
tab.tableStyleInfo = style
ws.add_table(tab)
tableList = ws.tables.items()   # 获取sheet中所有的table的表名和范围
del ws.tables["Table1"] # 删除表
wb.save("table.xlsx")

创建饼图 #

python
# 绘制饼图
import openpyxl
from openpyxl.chart import PieChart, Reference
from openpyxl.chart.series import DataPoint

# 准备数据
rows = [
    ['Pie', 'Sold'],
    ['Apple', 50],
    ['Cherry', 30],
    ['Pumpkin', 10],
    ['Chocolate', 40]
]

# 将数据写入excel
# 创建工作簿
wb = openpyxl.Workbook()
ws = wb.active
ws.title = 'Pie Charts'

for row in rows:
    ws.append(row)

# 绘制饼图
pie_chart = PieChart()
# 设置标题
pie_chart.title = 'Pie sold by category'
# 进行分类
category = Reference(ws, min_col=1, min_row=2, max_row=5)
data = Reference(ws, min_col=2, min_row=2, max_row=5)  # 数据所在第2列

# 需要先添加数据再设置种类介绍
# 添加数据
pie_chart.add_data(data)
# 设置所分类别
pie_chart.set_categories(category)
# 突出选中效果
slice = DataPoint(idx=0, explosion=20)
pie_chart.series[0].data_points = [slice]
# 在excel添加饼图
ws.add_chart(pie_chart, 'D1')  # 在D1位置绘制饼图
# 保存
wb.save('char_excel_text.xlsx')

创建柱状图 #

python
import openpyxl
from openpyxl.chart import PieChart, Reference, BarChart, BubbleChart, ScatterChart  # Reference：图标所用信息
from openpyxl.chart import Series

wb = openpyxl.Workbook()
ws = wb.active
# 绘制柱状图
# 创建工作表
ws = wb.create_sheet('Bar Chart')

# 准备数据
rows = [
    ('Number', 'Batch1', 'Batch2'),
    (2, 10, 30),
    (3, 40, 60),
    (4, 50, 70),
    (5, 20, 10),
    (6, 10, 40),
    (7, 50, 30),
]

# 添加数据
for row in rows:
    ws.append(row)

# 绘制柱状图
bar_chart = BarChart()
bar_chart.type = 'col'  # col垂直、水平柱状图 bar
bar_chart.title = 'Bar Chart'
bar_chart.style = 10  # 设置颜色，10的对比度最强，红色与蓝色
# 设置横轴纵轴标题
bar_chart.x_axis.title = 'Sample length(mm)'
bar_chart.y_axis.title = 'Test number'

# 设置分类
category = Reference(ws, min_col=1, min_row=2, max_row=7)
# 获取数据
data = Reference(ws, min_col=2, max_col=3, min_row=1, max_row=7)

# 柱状图对象添加数据
bar_chart.add_data(data, titles_from_data=True)  # titles_from_data=True：根据来源设置数据标题
# 设置分类
bar_chart.set_categories(category)
# 工作页绘制柱状图，并指定位置
ws.add_chart(bar_chart, 'E1')

# 保存
wb.save('char_excel_text.xlsx')

创建气泡图 #

python
import openpyxl
from openpyxl.chart import PieChart, Reference, BarChart, BubbleChart, ScatterChart  # Reference：图标所用信息
from openpyxl.chart import Series

# 绘制气泡图
ws = wb.create_sheet('Bubble Chart')

# 设置数据
rows = [
    ('Number of Products', 'Sales in USD', 'Marked share'),
    (14, 12200, 15),
    (20, 60000, 33),
    (18, 2440, 10),
    (22, 3200, 42),
    (),
    (12, 8200, 18),
    (15, 50000, 30),
    (19, 22400, 15),
    (25, 25000, 50),
]

# 添加数据
for row in rows:
    ws.append(row)

# 获取气泡图对象
bubble_chart = BubbleChart()
bubble_chart.style = 10  # 设置颜色

# 添加一组数据
xvalues = Reference(ws, min_col=1, min_row=2, max_row=5)
yvalues = Reference(ws, min_col=2, min_row=2, max_row=5)
size = Reference(ws, min_col=3, min_row=2, max_row=5)
# 创建Series对象
series = Series(values=yvalues, xvalues=xvalues, zvalues=size, title='2013')
bubble_chart.series.append(series)

# 添加一组数据
xvalues = Reference(ws, min_col=1, min_row=7, max_row=10)
yvalues = Reference(ws, min_col=2, min_row=7, max_row=10)
size = Reference(ws, min_col=3, min_row=7, max_row=10)
series = Series(values=yvalues, xvalues=xvalues, zvalues=size, title='2014')
bubble_chart.series.append(series)

# 添加气泡表
ws.add_chart(bubble_chart, 'E1')
# 保存
wb.save('char_excel_text.xlsx')

创建散点图 #

python
import openpyxl
from openpyxl.chart import PieChart, Reference, BarChart, BubbleChart, ScatterChart  # Reference：图标所用信息
from openpyxl.chart import Series

# 绘制散点图
ws = wb.create_sheet('Scatter Chart')

rows = [
    ['Size', 'Batch1', 'Batch2'],
    [2, 40, 25],
    [3, 40, 25],
    [4, 50, 30],
    [5, 30, 25],
    [6, 25, 35],
    [7, 20, 40],
]

for row in rows:
    ws.append(row)

# 绘制散点图
scatter_chart = ScatterChart()
# 设置标题
scatter_chart.title = 'Scatter Chart'
# 设置颜色
scatter_chart.style = 13

# 设置x轴y轴标题
scatter_chart.x_axis.title = 'Size'
scatter_chart.y_axis.title = 'Percentage'

# 创建x轴的数据来源
xvalues = Reference(ws, min_col=1, min_row=2, max_row=7)
# 创建yvalues
for i in range(2, 4):
    yvalues = Reference(ws, min_col=i, min_row=1, max_row=7)
    series = Series(yvalues, xvalues=xvalues, title_from_data=True)
    scatter_chart.series.append(series)

# 将散点图添加到ws工作表中
ws.add_chart(scatter_chart, 'E1')

# 保存工作簿
wb.save('charts.xlsx')

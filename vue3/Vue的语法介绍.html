Vue模板语法 #

  • v-html：使用 v-html 指令用于在div中输出 html 代码
  • v-bind：用于绑定div的属性，如id,class等等。完整写法v-bind:id="变量名",可以缩写成:id="变量名"。
  • v-if：判断。
  • v-for：循环
  • v-on：事件监听，完整的写法v-on:click="方法名",可以缩写为@click="方法名"
  • v-model ：绑定数据。
  • v-show：控制是否展示某元素，写法v-show="bool"

Vue的钩子函数 #

  • created : 在绑定元素的属性或事件监听器被应用之前调用。
  • beforeMount : 指令第一次绑定到元素并且在挂载父组件之前调用。。
  • mounted : 在绑定元素的父组件被挂载后调用。。
  • beforeUpdate: 在更新包含组件的 VNode 之前调用。
  • updated: 在包含组件的 VNode 及其子组件的 VNode 更新后调用。
  • beforeUnmount: 当指令与在绑定元素父组件卸载之前时，只调用一次。
  • unmounted: 当指令与元素解除绑定且父组件已卸载时，只调用一次。

Vue的合成API代码编写结构 #

JavaScript
<script>
import {ref,reactive,torefs,computed,watchEffect,watch} from 'vue'
export default {
    setup(){
        const count = ref(0);  //定义一个变量需要用ref来创建。
        const user = reactive({  //定义对象需要用reactive来创建。
            userName:"张三",
            age:30,
            type:"学生",
            reType:computed(()=>{    //计算属性的使用
                return user.type.lenght;
            })
        })
        function changeCount(){
            count.value++    //操作变量的值，需要用到这个变量的value属性
        }
        function changeUser(){
            user.type = '老师'
        }
        watchEffect(()=>{
            console.log(user.type)  //监听方法，自动绑定监听对象，当user.type发生改变时，会触发此方法。
        })
        //单个监听
        watch(count,(newvalue,oldvalue)=>{  //新的值、旧的值
            console.log(user.type)  //监听方法，指定监听某对象，当count发生改变时，会触发此方法。
        })
        //多个监听
        watch([count,user],(newvalue,oldvalue)=>{  //新的值、旧的值
            console.log(user.type)  //监听方法，指定监听某对象，当count发生改变时，会触发此方法。
        })
        return {   //不管是定义的变量还是方法，都需要用return导出
            count,
            changeCount,
            user,
            ...torefs(user),  //torefs可以将对象里面的值解析成单独的变量
            changeUser
        }
    }
}
</script>

JavaScript
<template>
    <div>
        <h1>username:{{username}}</h1>
        <h1>age:{{age}}</h1>
        <h1>des:{{des}}</h1>
        <butten @click=redes>返回des的值</butten>
    </div>
</template>
<script>
import {ref} rom 'vue'
export default {
    setup(props){
        const des = ref(props.username + props.age)
        function redes(){
            this.$emit('getdes',des)   //自定义事件,子组件操作父组件中的方法
        }
        return {
            des,redes
        }
    }
    props:['username','age']  //接受父组件传入的参数
}
</script>
